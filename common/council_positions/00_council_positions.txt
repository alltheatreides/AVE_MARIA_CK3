	modifier = {

	council_owner_modifier = {
		name = learn_on_the_job_modifier
		diplomacy = 1
		scale = chancellor_learn_on_the_job_scale
	}

	valid_character = {
		exists = root.liege_or_court_owner
		can_be_chancellor_trigger = { COURT_OWNER = root.liege_or_court_owner }
		trigger_if = {
			limit = {
				root.liege_or_court_owner = {
					uses_roman_govt = yes
					highest_held_title_tier = tier_empire
				}
			}
			is_ruler = yes
			has_primary_title = title:d_logothetetes_tou_dromou
		}
		## AVE MARIA Muslim triggers
		trigger_if = {
			limit = {
				root.liege_or_court_owner = { 
					uses_islamic_empire_govt = yes
					highest_held_title_tier = tier_empire
					any_held_title = {
						has_variable = islamic_empire_title
					}
				}
			}
			NOT = {
				has_government = theocracy_government
			}
			trigger_if = {
				limit = {
					root.liege_or_court_owner = {
						OR = {
							has_character_modifier = islamic_empire_faction_ascendency_high_army
							has_character_modifier = islamic_empire_faction_ascendency_normal_army
						}					
					}
				}
				OR = {
					has_government = islamic_empire_regiment_government
					AND = {
						is_landed = yes
						highest_held_title_tier >= tier_duchy
						martial >= 12
					} 
				}
			}
			trigger_if = {
				limit = {
					root.liege_or_court_owner = {
						OR = {
							has_character_modifier = islamic_empire_faction_ascendency_high_diwan
							has_character_modifier = islamic_empire_faction_ascendency_normal_diwan
						}					
					}
				}
				OR = {
					AND = {
						is_ruler = yes
						has_government = islamic_empire_diwan_government
					}
					is_ruler = no
				}
				
			}
			trigger_if = {
				limit = {
					root.liege_or_court_owner = {
						has_character_modifier = islamic_empire_faction_influential_diwan_faction
					}
				}
				trigger_if = {
					limit = {
						has_government = islamic_empire_diwan_government
					}
					primary_title = {
						has_variable = islamic_empire_privilege_position
					}
				}
			}
			# trigger_if = {
			# 	limit = {
			# 		root.liege_or_court_owner = {
			# 			has_character_flag = insha_ascendant
			# 		}
			# 	}
			# }
		}
	}

	on_get_position = {
		got_council_position_effect = yes
	}

	can_fire = {
		trigger_if = {
			limit = {
				liege = {
					has_government = islamic_empire_government
					highest_held_title_tier = tier_empire

			}
			NOT = {
