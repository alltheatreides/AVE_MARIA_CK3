am_change_govt_if_contract_changed_effect = {
	if = {
		limit = {
			local_var:$AM_GOVT_TYPE$ != vassal_contract_obligation_level:$AM_GOVT_TYPE$
		}
		if = {
			limit = {
				#vassal_contract_obligation_level:$AM_GOVT_TYPE$ = ave_maria_roman_administration_fiscalized_theme
				vassal_contract_has_flag = is_theme
			}
			change_government = roman_administration_theme_government
		}
		else_if = {
			limit = {
				#vassal_contract_obligation_level:$AM_GOVT_TYPE$ = ave_maria_roman_administration_theme
				vassal_contract_has_flag = is_theme
			}
			change_government = roman_administration_theme_government
		}
		else_if = {
			limit = {
				#vassal_contract_obligation_level:$AM_GOVT_TYPE$ = ave_maria_roman_administration_doukaton
				vassal_contract_has_flag = is_doukaton
			}
			change_government = roman_administration_doukaton_government
		}
		else_if = {
			limit = {
				vassal_contract_obligation_level:$AM_GOVT_TYPE$ = ave_maria_roman_administration_kouratoreia
				vassal_contract_has_flag = is_kouratoreia
			}
			change_government = roman_administration_kouratoreia_government
		}
	}
}

roman_administration_update_government_after_contract_change = {
	if = {
		limit = {
			vassal_contract_has_flag = is_theme
		}
		NOT = { has_government = roman_administration_theme_government }
		change_government = roman_administration_theme_government
	}
	if = {
		limit = {
			vassal_contract_has_flag = is_doukaton
		}
		NOT = { has_government = roman_administration_doukaton_government }
		change_government = roman_administration_doukaton_government
	}
	if = {
		limit = {
			vassal_contract_has_flag = is_kouratoreia
		}
		NOT = { has_government = roman_administration_kouratoreia_government }
		change_government = roman_administration_kouratoreia_government
	}
	if = {
		limit = {
			vassal_contract_has_flag = is_tagma
		}
		NOT = { has_government = roman_administration_military_government }
		change_government = roman_administration_military_government
	}
	if = {
		limit = {
			vassal_contract_has_flag = is_sekreton
		}
		NOT = { has_government = roman_administration_sekreton_government }
		change_government = roman_administration_sekreton_government
	}
}

roman_administration_emperor_inheritance_effect = {
	## Removes terms of office if you were an Official before
	roman_administration_remove_terms_of_office_modifier_and_variable = yes	
	
	## Remove the junior emperor trait
	remove_trait = junior_emperor

	#Proper government assignement
	change_government = roman_administration_government
	
	## Proper low assignement
	add_realm_law = ave_maria_roman_administration_imperial
	
	## Give you the Salary malus
	roman_administration_update_total_salary_cost = yes

	## Updating Legitimacy and Popularity
	roman_administration_update_popularity_variable = yes
	roman_administration_update_legitimacy_variable = yes

	## Giving Constantinople to the new emperor
	title:c_byzantion = {
		create_title_and_vassal_change = {
			type = granted
			save_scope_as = change
			add_claim_on_loss = yes
		}
		save_temporary_scope_as = title_to_inherit
		root = {
			save_scope_as = title_heir
		}
		change_title_holder = {
			holder = scope:title_heir
			change = scope:change
		}
	}
	resolve_title_and_vassal_change = scope:change

}

roman_administration_civilian_titles_transfer = {
	## Flagging the duchy administrative title heir
	every_held_title = {
		limit = {
			tier = tier_duchy
		}
		current_heir = {
			save_scope_as = title_heir
		}
	}
	## Transferring counties to stick the transfer
	every_held_title = {
		limit = {
			tier = tier_county
			NOT = { this = title:e_byzantium }
			NOT = {
				this = { has_title_law = ave_maria_roman_administration_family_succession_law }	
				has_title_law = ave_maria_roman_administration_family_succession_law
			}
			NOT = { this = title:c_byzantion }
		}
		create_title_and_vassal_change = {
			type = granted
			save_scope_as = change
			add_claim_on_loss = yes
		}
		save_temporary_scope_as = title_to_inherit
		change_title_holder = {
			holder = scope:title_heir
			change = scope:change
		}
		resolve_title_and_vassal_change = scope:change
	}

	every_held_title = {

			tier = tier_duchy
		}

		}

			save_scope_as = title_heir
		}

			holder = scope:title_heir

		}

	}		
	
	resolve_title_and_vassal_change = scope:change

## Update calculations of Popularity variable
roman_administration_update_popularity_variable = {

	## Creating a calculation variable for operations
	## First adding static Martial, Stewardship, Prowess, Prestige and Piety
	set_variable = {
		name = calculate_character_popularity
		value = 0
	}
	## Dividing Prestige Value by 10 as prestige values are pretty big
	set_variable = {
		name = calculate_character_popularity_prestige
		value = 0
	}
	change_variable = {
		name = calculate_character_popularity_prestige
		add = prestige 
	}
	change_variable = {
		name = calculate_character_popularity_prestige
		multiply = 0.1 
	}
	# Adding the result to popularity variable
	change_variable = {
		name = calculate_character_popularity
		add = var:calculate_character_popularity_prestige
	}
	remove_variable = calculate_character_popularity_prestige

	## Dividing Piety Value by 10 as piety values are pretty big
	set_variable = {
		name = calculate_character_popularity_piety
		value = 0
	}
	change_variable = {
		name = calculate_character_popularity_piety
		add = piety 
	}
	change_variable = {
		name = calculate_character_popularity_piety
		multiply = 0.1 
	}
	# Adding the result to popularity variable
	change_variable = {
		name = calculate_character_popularity
		add = var:calculate_character_popularity_piety
	}
	remove_variable = calculate_character_popularity_piety
	
	## Straight up adding Martial, Stewardship, Prowess
	change_variable = {
		name = calculate_character_popularity
		add = martial
	}
	change_variable = {
		name = calculate_character_popularity
		add = stewardship
	}
	change_variable = {
		name = calculate_character_popularity
		add = prowess
	}

	## Then static removal of points from traits (for now 5, might need to balance higher ?)
	if = {
		limit = {
			has_trait = am_inconstant
		}
		change_variable = {
			name = calculate_character_popularity
			add = -5
		}
	}
	if = {
		limit = {
			has_trait = am_hypocritical
		}
		change_variable = {
			name = calculate_character_popularity
			add = -5
		}
	}
	if = {
		limit = {
			has_trait = am_greedy
		}
		change_variable = {
			name = calculate_character_popularity
			add = -5
		}
	}
	if = {
		limit = {
			has_trait = am_intolerant
		}
		change_variable = {
			name = calculate_character_popularity
			add = -5
		}
	}

	## Add the result of calculation to the popularity variable if you don't
	if = {
		limit = {
			NOT = {
				has_variable = roman_administration_popularity
			}
		}
		## If no Popularity variable already, create it
		roman_administration_set_popularity_variable = yes
		change_variable = {
			name = roman_administration_popularity
			add = var:calculate_character_popularity
		}
	}


	## If Popularity variable already exists , compare result of calculation with current value and update
	if = {
		limit = {
			has_variable = roman_administration_popularity
		}

		## Replace new value with the other
		set_variable = {
			name = roman_administration_popularity
			value = var:calculate_character_popularity
		}
	}

	## Update the display variable
	roman_administration_update_popularity_level_display = yes

	## Clean up the variables
	remove_variable = calculate_character_popularity

}

	## At least one emperor in the family
	if = {
		limit = {
			has_dynasty = yes
			dynasty = {
				has_variable = roman_administration_dynastic_number_of_emperors
				var:roman_administration_dynastic_number_of_emperors >= 1
			}
		}
		change_variable = {
			name = calculate_character_legitimacy
			add = 50
		}
		set_variable = {
			name = character_legitimacy_display_dynastic_emperor_1
			value = 50
		}
	}
	## more than 5 emperos in the dynasty
	if = {
		limit = {
			has_dynasty = yes
			dynasty = {
				has_variable = roman_administration_dynastic_number_of_emperors
				var:roman_administration_dynastic_number_of_emperors >= 5
			}
		}
		change_variable = {
			name = calculate_character_legitimacy
			add = 100
		}
		set_variable = {
			name = character_legitimacy_display_dynastic_emperor_5
			value = 100
		}
	}
	## Direct child of the emperor
	if = {
		limit = {
			title:e_byzantium.holder = {
				is_parent_of = prev
			}
		}
		change_variable = {
			name = calculate_character_legitimacy
			add = 25
		}
	}
	
	## REMOVAL
	## Known murderer, the removal of legitimacy effect from murdering someone from imperial family should go in murder scheme effects with a trigger if they are of the imperial family
	if = {
		limit = {
			has_trait = murderer
		}
		change_variable = {
			name = calculate_character_legitimacy
			add = -25
		}
		set_variable = {
			name = character_legitimacy_display_murderer
			value = -25
		}
	}
	## Married to a non orthodox character
	if = {
		limit = {
			any_spouse = {
				NOT = { faith = faith:orthodox }
			}
		}
		change_variable = {
			name = calculate_character_legitimacy
			add = -25
		}
		set_variable = {
			name = character_legitimacy_display_wrong_religion
			value = -25
		}
	}
	## Married to a non roman character
	if = {
		limit = {
			any_spouse = {
				NOT = { culture = culture:greek }
			}
		}
		change_variable = {
			name = calculate_character_legitimacy
			add = -15
		}
		set_variable = {
			name = character_legitimacy_display_wrong_culture
			value = -15
		}
	}

	if = {
		limit = {
			OR = {
				has_trait = eunuch
				has_trait = infirm
				has_trait = maimed
			}
		}
		change_variable = {
			name = calculate_character_legitimacy
			add = -100
		}
	}

	## Add the result of calculation to the legitimacy variable if you don't
	if = {
		limit = {
			NOT = {
				has_variable = roman_administration_legitimacy
			}
		}
		## If no legitimacy variable already, create it
		roman_administration_set_legitimacy_variable = yes
		change_variable = {
			name = roman_administration_legitimacy
			add = var:calculate_character_legitimacy
		}
	}


	## If legitimacy variable already exists , compare result of calculation with current value and update
	if = {
		limit = {
			has_variable = roman_administration_legitimacy
		}

		## Replace new value with the other
		set_variable = {
			name = roman_administration_legitimacy
			value = var:calculate_character_legitimacy
		}
	}

	roman_administration_update_legitimacy_level_display = yes

	## Clean up the variables
	remove_variable = calculate_character_legitimacy

}

			value = 0
		}
	}

## Character Scope effect to display correct Legitimacy Level
roman_administration_update_legitimacy_level_display = {
	if = { 					
		limit = {				
			has_variable = roman_administration_legitimacy			
			var:roman_administration_legitimacy > 0			
			var:roman_administration_legitimacy <= 50			
		}				
		set_variable = {				
			name = roman_administration_legitimacy_level_display			
			value = flag:despised			
		}				
	}	
	if = { 					
		limit = {				
			has_variable = roman_administration_legitimacy			
			var:roman_administration_legitimacy > 50
			var:roman_administration_legitimacy <= 100			
		}				
		set_variable = {				
			name = roman_administration_legitimacy_level_display			
			value = flag:unworthy			
		}				
	}	
	if = { 					
		limit = {				
			has_variable = roman_administration_legitimacy			
			var:roman_administration_legitimacy > 100
			var:roman_administration_legitimacy <= 150			
		}				
		set_variable = {				
			name = roman_administration_legitimacy_level_display			
			value = flag:tolerated			
		}				
	}	
	if = { 					
		limit = {				
			has_variable = roman_administration_legitimacy			
			var:roman_administration_legitimacy > 200
			var:roman_administration_legitimacy <= 250			
		}				
		set_variable = {				
			name = roman_administration_legitimacy_level_display			
			value = flag:worthy			
		}				
	}	
	if = { 					
		limit = {				
			has_variable = roman_administration_legitimacy			
			var:roman_administration_legitimacy >= 300		
		}				
		set_variable = {				
			name = roman_administration_legitimacy_level_display			
			value = flag:beloved			
		}				
	}			
	
}
