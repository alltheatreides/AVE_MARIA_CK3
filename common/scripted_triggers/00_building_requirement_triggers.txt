######################################################################
# Government
######################################################################

building_requirement_tribal = {
	scope:holder = {
		
		OR = {
			has_government = tribal_government
			AND = {
				has_government = clan_government
				religion = religion:islam_religion
			}
			AND = {
				has_government = clan_government
				capital_county = {
					title_province = {
						OR = {
							has_holding_type = tribal_holding
							has_holding_type = pastoral_holding
						}
					}
				}
			}
		}
		#always = yes
	}
}

building_requirement_tribal_holding_in_county = {
	county = {
		any_county_province = {
			OR = {
				has_holding_type = tribal_holding
				AND = {
					province_owner = { has_government = tribal_government }
					has_holding_type = pastoral_holding
				}
				
			}			
		}
	}
}

######################################################################
# Holding Level
######################################################################

building_requirement_castle_city_church = {
	trigger_if = {
		limit = { has_holding_type = castle_holding }
		has_building_or_higher = castle_$LEVEL$
	}
	trigger_else_if = {
		limit = { has_holding_type = city_holding }
		has_building_or_higher = city_$LEVEL$
	}
	trigger_else_if = {
		limit = { has_holding_type = church_holding }
		has_building_or_higher = temple_$LEVEL$
	}
	trigger_else_if = {
		limit = { has_holding_type = metropolis_constantinople_holding }
		has_building = metropolis_constantinople_01
	}
	trigger_else_if = {
		limit = { has_holding_type = pastoral_holding }
		province_owner = { NOT = { has_government = tribal_government } }
	}
	trigger_else = {
		custom_description = {
			text = "building_requirement_castle_city_church_failed"
			always = no
		}
	}
}
